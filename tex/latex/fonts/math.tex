\ifundef{\myfont}{
  \newcommand{\myfont}{Latin Modern Roman}
}

\ifundef{\mymonofont}{
  \newcommand{\mymonofont}{Ubuntu Mono}
}

\usepackage{mathspec}
  \IfFontExistsTF{\myfont}{%
    \setmainfont{\myfont}[%
      Mapping=tex-text ,
    ]
    \setmathrm{\myfont}
    \setmathfont(Digits,Latin){\myfont}
  }{}
  \IfFontExistsTF{\mymonofont}{%
    \setmonofont[Scale=0.8]{\mymonofont}
  }{}
