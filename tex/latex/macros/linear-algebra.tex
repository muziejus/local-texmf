\newcounter{vecncounter}
\newbool{vecnlength}
\newcommand{\iteratevecn}[2]{
  \defcounter{vecncounter}{0}
  \unlessboolexpr{bool {vecnlength}}{
    \stepcounter{vecncounter}
    \ifnumcomp{\thevecncounter}{>}{#2}
      {\setbool{vecnlength}{true}}
      {
        \ifnumcomp{\thevecncounter}{=}{#2}
          {{#1}_{\thevecncounter}}
          {{#1}_{\thevecncounter} \\ }
      }
  }% end \unlessboolexpr
}% end \iteratevecn
